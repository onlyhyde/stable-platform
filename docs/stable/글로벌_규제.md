## 1. 일본 (Japan)

### 1.1 규제 요건

**규제 근거**: 자금결제법 (Payment Services Act) 개정  
**시행일**: 2023년 6월  
**감독 기관**: FSA (금융청)

#### 발행자 요건

| 항목 | 요건 |

| **외국 스테이블코인** | 국내 중개업자가 동일 금액 준비금 예치 필요 |

#### 준비금 요건

| **허용 자산** | 요구불예금, 단기국채 (3개월 이하), 조기해지 가능 정기예금 |

| **AML/KYC** | FSA 기준 준수 |

### 1.2 규제 준수 기술

#### 1.2.1 발행자 제한 시스템 (Minter Authorization)

**목적**: 은행, 자금이동업자, 신탁회사만 발행 가능하도록 제한

**기술 구현**:

```solidity
// MinterRegistry.sol
contract MinterRegistry {
    enum MinterType { BANK, MONEY_TRANSFER, TRUST_COMPANY, NONE }
    
    struct MinterInfo {
        address minter;
        MinterType minterType;
        bool isAuthorized;
        uint256 authorizedAmount;
        uint256 mintedAmount;
    }
    
    mapping(address => MinterInfo) public minters;
    address public govMasterMinter;
    
    modifier onlyAuthorizedMinter() {
        require(
            minters[msg.sender].isAuthorized && 
            minters[msg.sender].minterType != MinterType.NONE,
            "Not authorized minter"
        );
        _;
    }
    
    function registerMinter(
        address minter,
        MinterType minterType,
        uint256 maxAmount
    ) external onlyGovMasterMinter {
        require(
            minterType == MinterType.BANK ||
            minterType == MinterType.MONEY_TRANSFER ||
            minterType == MinterType.TRUST_COMPANY,
            "Invalid minter type"
        );
        
        minters[minter] = MinterInfo({
            minter: minter,
            minterType: minterType,
            isAuthorized: true,
            authorizedAmount: maxAmount,
            mintedAmount: 0
        });
    }
}
```

**구현 요건**:
- GovMasterMinter를 통한 발행자 등록
- 발행자 유형 검증 (은행/자금이동업자/신탁회사)
- 발행 한도 관리
- KYB (Know Your Business) 검증 연동

#### 1.2.2 준비금 관리 시스템 (Reserve Management)

**목적**: 100% 백업, 국채 비율 50% 이하, 분리 보관

**기술 구현**:

```solidity
// ReserveManager.sol
contract ReserveManager {
    struct ReserveAsset {
        address assetAddress;  // 0x0 for JPY
        uint256 amount;
        AssetType assetType;
    }
    
    enum AssetType { 
        DEMAND_DEPOSIT,      // 요구불예금
        SHORT_TERM_BOND,     // 단기국채 (3개월 이하)
        TIME_DEPOSIT         // 조기해지 가능 정기예금
    }
    
    mapping(AssetType => uint256) public reserveByType;
    uint256 public totalReserve;
    uint256 public totalSupply;
    uint256 public constant MAX_BOND_RATIO = 50; // 50%
    
    function updateReserve(
        AssetType assetType,
        uint256 amount,
        bool isIncrease
    ) external onlyAuthorized {
        if (isIncrease) {
            reserveByType[assetType] += amount;
            totalReserve += amount;
        } else {
            reserveByType[assetType] -= amount;
            totalReserve -= amount;
        }
        
        // 국채 비율 검증
        uint256 bondRatio = (reserveByType[AssetType.SHORT_TERM_BOND] * 100) / totalReserve;
        require(bondRatio <= MAX_BOND_RATIO, "Bond ratio exceeds limit");
        
        // 100% 백업 검증
        require(totalReserve >= totalSupply, "Reserve insufficient");
    }
}
```

**구현 요건**:
- 실시간 준비금 추적 (은행 계좌 API 연동)
- 자산 유형별 분류 및 관리
- 국채 비율 자동 검증
- 100% 백업 실시간 검증

#### 1.2.3 분리 보관 시스템 (Segregated Custody)

**목적**: 발행자 자산과 준비금 완전 분리

**기술 구현**:

```solidity
// SegregatedTreasury.sol
contract SegregatedTreasury {
    address public reserveAccount;
    address public operationalAccount;
    address public custodian;
    
    modifier onlyCustodian() {
        require(msg.sender == custodian, "Not custodian");
        _;
    }
    
    function depositReserve(uint256 amount) external payable {
        require(msg.value == amount, "Amount mismatch");
        // 준비금 계좌로만 입금 가능
        reserveAccount.transfer(amount);
    }
    
    function withdrawReserve(
        address to,
        uint256 amount
    ) external onlyCustodian {
        require(
            address(this).balance >= amount,
            "Insufficient reserve"
        );
        to.transfer(amount);
    }
}
```

**구현 요건**:
- 준비금 전용 계좌 분리
- 운영 자산과 완전 분리된 구조
- 수탁사(Custodian) 다중 서명
- 오프라인 보관 95% 이상 (콜드 월렛)

#### 1.2.4 연간 감사 시스템 (Annual Audit)

**목적**: 연 1회 외부 감사 필수

**기술 구현**:

```solidity
// AuditManager.sol
contract AuditManager {
    struct AuditReport {
        uint256 auditDate;
        address auditor;
        uint256 totalReserve;
        uint256 totalSupply;
        bool isCompliant;
        string reportHash; // IPFS hash
    }
    
    AuditReport[] public auditReports;
    uint256 public lastAuditDate;
    uint256 public constant AUDIT_INTERVAL = 365 days;
    
    function submitAuditReport(
        uint256 totalReserve,
        uint256 totalSupply,
        bool isCompliant,
        string memory reportHash
    ) external onlyAuthorizedAuditor {
        require(
            block.timestamp >= lastAuditDate + AUDIT_INTERVAL,
            "Audit too frequent"
        );
        
        auditReports.push(AuditReport({
            auditDate: block.timestamp,
            auditor: msg.sender,
            totalReserve: totalReserve,
            totalSupply: totalSupply,
            isCompliant: isCompliant,
            reportHash: reportHash
        }));
        
        lastAuditDate = block.timestamp;
    }
}
```

**구현 요건**:
- 외부 감사 기관 등록
- 감사 보고서 온체인 저장
- IPFS를 통한 상세 보고서 저장
- 감사 주기 자동 알림

### 2.2 규제 준수 기술

#### 2.2.1 월간 검증 시스템 (Monthly Attestation)

**목적**: 월 1회 독립 검증 및 공개

**기술 구현**:

```solidity
// MonthlyAttestation.sol
contract MonthlyAttestation {
    struct AttestationReport {
        uint256 reportDate;
        address attestor;
        uint256 totalReserve;
        uint256 totalSupply;
        uint256 reserveByCurrency;
        bool isCompliant;
        string reportHash; // IPFS hash
    }
    
    AttestationReport[] public monthlyReports;
    uint256 public lastAttestationDate;
    uint256 public constant ATTESTATION_INTERVAL = 30 days;
    
    mapping(address => bool) public authorizedAttestors;
    
    modifier onlyAuthorizedAttestor() {
        require(authorizedAttestors[msg.sender], "Not authorized attestor");
        _;
    }
    
    function submitMonthlyAttestation(
        uint256 totalReserve,
        uint256 totalSupply,
        uint256 reserveByCurrency,
        bool isCompliant,
        string memory reportHash
    ) external onlyAuthorizedAttestor {
        require(
            block.timestamp >= lastAttestationDate + ATTESTATION_INTERVAL,
            "Attestation too frequent"
        );
        
        require(
            totalReserve >= totalSupply,
            "Reserve insufficient"
        );
        
        monthlyReports.push(AttestationReport({
            reportDate: block.timestamp,
            attestor: msg.sender,
            totalReserve: totalReserve,
            totalSupply: totalSupply,
            reserveByCurrency: reserveByCurrency,
            isCompliant: isCompliant,
            reportHash: reportHash
        }));
        
        lastAttestationDate = block.timestamp;
        
        emit MonthlyAttestationSubmitted(
            block.timestamp,
            msg.sender,
            isCompliant
        );
    }
    
    function getLatestAttestation() 
        external 
        view 
        returns (AttestationReport memory) 
    {
        require(monthlyReports.length > 0, "No attestation");
        return monthlyReports[monthlyReports.length - 1];
    }
}
```

**구현 요건**:
- 등록 회계법인 또는 인증된 검증 기관 연동
- 월간 자동 검증 스케줄링
- IPFS를 통한 상세 보고서 저장
- 공개 대시보드 자동 업데이트

#### 2.2.2 일일 평가 시스템 (Daily Mark-to-Market)

**목적**: 준비금 일일 시가 평가

**기술 구현**:

```solidity
// DailyValuation.sol
contract DailyValuation {
    struct DailyValuationData {
        uint256 valuationDate;
        uint256 totalReserveValue;
        uint256 totalSupply;
        uint256 reserveByAssetType;
        bool isCompliant;
    }
    
    mapping(uint256 => DailyValuationData) public dailyValuations;
    address public priceOracle;
    
    function updateDailyValuation(
        uint256 reserveByAssetType,
        uint256 totalSupply
    ) external onlyAuthorized {
        uint256 reserveValue = calculateReserveValue(reserveByAssetType);
        
        dailyValuations[block.timestamp / 1 days] = DailyValuationData({
            valuationDate: block.timestamp,
            totalReserveValue: reserveValue,
            totalSupply: totalSupply,
            reserveByAssetType: reserveByAssetType,
            isCompliant: reserveValue >= totalSupply
        });
        
        require(
            reserveValue >= totalSupply,
            "Daily valuation: Reserve insufficient"
        );
    }
    
    function calculateReserveValue(uint256 reserveByAssetType) 
        internal 
        view 
        returns (uint256) 
    {
        // Price Oracle을 통한 시가 평가
        // 현금: 1:1
        // 현금성 자산: 시가 평가
        // 국채: 시가 평가 (3개월 이하)
        return IPriceOracle(priceOracle).getTotalValue(reserveByAssetType);
    }
}
```

**구현 요건**:
- Price Oracle 연동 (Chainlink 등)
- 일일 자동 평가 스케줄링
- 시가 평가 로직 구현
- 비준수 시 자동 알림

#### 2.2.3 통화 일치 검증 시스템 (Currency Matching)

**목적**: 페그 통화와 동일 통화로 준비금 보유

**기술 구현**:

```solidity
// CurrencyMatching.sol
contract CurrencyMatching {
    enum PegCurrency { SGD, USD, EUR, JPY, GBP, AUD, CAD, CHF, NZD, SEK }
    
    mapping(PegCurrency => uint256) public reserveByCurrency;
    PegCurrency public peggedCurrency;
    uint256 public totalSupply;
    
    function updateReserve(
        PegCurrency currency,
        uint256 amount,
        bool isIncrease
    ) external onlyAuthorized {
        if (isIncrease) {
            reserveByCurrency[currency] += amount;
        } else {
            reserveByCurrency[currency] -= amount;
            require(
                reserveByCurrency[currency] >= 0,
                "Insufficient reserve"
            );
        }
        
        // 페그 통화와 일치 검증
        require(
            reserveByCurrency[peggedCurrency] >= totalSupply,
            "Currency mismatch: Reserve must match pegged currency"
        );
    }
    
    function getReserveRatio() 
        external 
        view 
        returns (uint256) 
    {
        if (totalSupply == 0) return 0;
        return (reserveByCurrency[peggedCurrency] * 100) / totalSupply;
    }
}
```

**구현 요건**:
- 페그 통화별 준비금 추적
- 통화 일치 실시간 검증
- 다중 통화 지원 (G10 통화)

#### 2.2.4 상환 관리 시스템 (Redemption Management)

**목적**: 액면가 상환, 5영업일 이내 처리

**기술 구현**:

```solidity
// RedemptionManager.sol
contract RedemptionManager {
    struct RedemptionRequest {
        address requester;
        uint256 amount;
        uint256 requestDate;
        uint256 deadline; // 5 영업일
        bool isProcessed;
    }
    
    RedemptionRequest[] public redemptionQueue;
    mapping(address => uint256[]) public userRequests;
    
    uint256 public constant MAX_REDEMPTION_DAYS = 5;
    
    function requestRedemption(uint256 amount) external {
        require(amount > 0, "Amount must be positive");
        require(
            IStablecoin(stablecoinAddress).balanceOf(msg.sender) >= amount,
            "Insufficient balance"
        );
        
        // 토큰 소각
        IStablecoin(stablecoinAddress).burnFrom(msg.sender, amount);
        
        uint256 deadline = calculateBusinessDays(block.timestamp, MAX_REDEMPTION_DAYS);
        
        uint256 requestId = redemptionQueue.length;
        redemptionQueue.push(RedemptionRequest({
            requester: msg.sender,
            amount: amount,
            requestDate: block.timestamp,
            deadline: deadline,
            isProcessed: false
        }));
        
        userRequests[msg.sender].push(requestId);
        
        emit RedemptionRequested(msg.sender, amount, requestId);
    }
    
    function processRedemption(
        uint256 requestId,
        address to
    ) external onlyAuthorized {
        RedemptionRequest storage request = redemptionQueue[requestId];
        require(!request.isProcessed, "Already processed");
        require(block.timestamp <= request.deadline, "Deadline passed");
        
        // 준비금에서 상환
        require(
            address(this).balance >= request.amount,
            "Insufficient reserve"
        );
        
        payable(to).transfer(request.amount);
        request.isProcessed = true;
        
        emit RedemptionProcessed(requestId, to, request.amount);
    }
    
    function calculateBusinessDays(
        uint256 startDate,
        uint256 daysToAdd
    ) internal pure returns (uint256) {
        // 영업일 계산 로직 (주말 제외)
        // 실제 구현 시 공휴일 캘린더 연동 필요
        return startDate + (daysToAdd * 1 days);
    }
}
```

**구현 요건**:
- FIFO 상환 큐 시스템
- 영업일 계산 로직 (공휴일 제외)
- 5영업일 이내 처리 보장
- 액면가 1:1 상환

        TokenType tokenType;
        IssuerType issuerType;

            (tokenType == TokenType.EMT &&

#### 4.2.2 1영업일 상환 시스템 (1-Day Redemption)

**목적**: 1영업일 이내 상환 처리 (가장 엄격)

**기술 구현**:

```solidity
// FastRedemption.sol
contract FastRedemption {
    struct RedemptionRequest {
        address requester;
        uint256 amount;
        uint256 requestTimestamp;
        uint256 deadline; // 1 영업일
        bool isProcessed;
    }
    
    RedemptionRequest[] public redemptionQueue;
    uint256 public constant MAX_REDEMPTION_HOURS = 24; // 1 영업일
    
    function requestRedemption(uint256 amount) external {
        require(amount > 0, "Amount must be positive");
        
        // 토큰 소각
        IStablecoin(stablecoinAddress).burnFrom(msg.sender, amount);
        
        uint256 deadline = block.timestamp + (MAX_REDEMPTION_HOURS * 1 hours);
        
        uint256 requestId = redemptionQueue.length;
        redemptionQueue.push(RedemptionRequest({
            requester: msg.sender,
            amount: amount,
            requestTimestamp: block.timestamp,
            deadline: deadline,
            isProcessed: false
        }));
        
        emit RedemptionRequested(msg.sender, amount, requestId, deadline);
        
        // 자동 처리 트리거
        processRedemptionIfReady(requestId);
    }
    
    function processRedemption(uint256 requestId) external {
        RedemptionRequest storage request = redemptionQueue[requestId];
        require(!request.isProcessed, "Already processed");
        require(block.timestamp <= request.deadline, "Deadline passed");
        
        require(
            address(this).balance >= request.amount,
            "Insufficient reserve"
        );
        
        // 즉시 상환
        payable(request.requester).transfer(request.amount);
        request.isProcessed = true;
        
        emit RedemptionProcessed(requestId, request.requester, request.amount);
    }
    
    function processRedemptionIfReady(uint256 requestId) internal {
        RedemptionRequest storage request = redemptionQueue[requestId];
        if (
            !request.isProcessed &&
            block.timestamp <= request.deadline &&
            address(this).balance >= request.amount
        ) {
            processRedemption(requestId);
        }
    }
    
    // 자동 처리 크론잡
    function autoProcessRedemptions() external {
        for (uint256 i = 0; i < redemptionQueue.length; i++) {
            if (!redemptionQueue[i].isProcessed) {
                processRedemptionIfReady(i);
            }
        }
    }
}
```

**구현 요건**:
- 1영업일 이내 처리 보장
- 자동 상환 프로세스
- 실시간 모니터링
- 데드라인 알림

contract FeeRestriction {
    uint256 public maxRedemptionFee; // 최대 상환 수수료 (예: 0.1%)
    uint256 public maxMintFee; // 최대 발행 수수료
    uint256 public constant MAX_FEE_BASIS_POINTS = 10; // 0.1%
    
    mapping(address => uint256) public redemptionFees;
    mapping(address => uint256) public mintFees;

            feeBasisPoints <= MAX_FEE_BASIS_POINTS,
            "Fee exceeds maximum allowed"
        );
        maxRedemptionFee = feeBasisPoints;
    }
    
    function calculateRedemptionFee(uint256 amount) 
        public 
        view 
        returns (uint256) 
    {
        return (amount * maxRedemptionFee) / 10000;
    }
    
    function processRedemptionWithFee(
        address to,
        uint256 amount
    ) external onlyAuthorized {
        uint256 fee = calculateRedemptionFee(amount);
        uint256 netAmount = amount - fee;
        
        require(
            fee <= (amount * MAX_FEE_BASIS_POINTS) / 10000,

// NextDayRedemption.sol
contract NextDayRedemption {
    struct RedemptionRequest {
        address requester;
        uint256 amount;
        uint256 requestDate;
        uint256 deadline; // 익영업일 말
        bool isProcessed;
    }
    
    RedemptionRequest[] public redemptionQueue;
    
    function requestRedemption(uint256 amount) external {
        require(amount > 0, "Amount must be positive");
        
        IStablecoin(stablecoinAddress).burnFrom(msg.sender, amount);
        
        uint256 deadline = getNextBusinessDayEnd(block.timestamp);
        
        uint256 requestId = redemptionQueue.length;
        redemptionQueue.push(RedemptionRequest({
            requester: msg.sender,
            amount: amount,
            requestDate: block.timestamp,
            deadline: deadline,
            isProcessed: false
        }));
        
        emit RedemptionRequested(msg.sender, amount, requestId, deadline);
    }
    
    function getNextBusinessDayEnd(uint256 timestamp) 
        internal 
        pure 
        returns (uint256) 
    {
        // 익영업일 말 계산
        // 주말 및 공휴일 제외
        uint256 nextDay = timestamp + 1 days;
        
        // 주말 처리 (실제 구현 시 공휴일 캘린더 연동)
        uint256 dayOfWeek = (nextDay / 1 days) % 7;
        if (dayOfWeek == 0) { // 일요일
            nextDay += 1 days;
        } else if (dayOfWeek == 6) { // 토요일
            nextDay += 2 days;
        }
        
        // 익영업일 말 (23:59:59)
        return (nextDay / 1 days) * 1 days + 23 hours + 59 minutes + 59 seconds;
    }
    
    function processRedemption(uint256 requestId) external {
        RedemptionRequest storage request = redemptionQueue[requestId];
        require(!request.isProcessed, "Already processed");
        require(block.timestamp <= request.deadline, "Deadline passed");
        
        require(
            address(this).balance >= request.amount,
            "Insufficient reserve"
        );
        
        payable(request.requester).transfer(request.amount);
        request.isProcessed = true;
        
        emit RedemptionProcessed(requestId, request.requester, request.amount);
    }
}
```

// KoreanMinterRegistry.sol
contract KoreanMinterRegistry {
    enum MinterType { BANK, FINANCIAL_INSTITUTION, OTHER }
    
    struct MinterInfo {
        address minter;
        MinterType minterType;
        uint256 sharePercentage; // 지분 비율
        uint256 authorizedAmount;
        uint256 mintedAmount;
        bool isAuthorized;
    }
    
    mapping(address => MinterInfo) public minters;
    address[] public minterList;
    uint256 public totalBankShare; // 은행 지분 합계
    uint256 public constant MIN_BANK_SHARE = 51; // 51%
    
    function registerMinter(
        address minter,
        MinterType minterType,
        uint256 sharePercentage,
        uint256 maxAmount
    ) external onlyGovMasterMinter {
        require(
            minterType == MinterType.BANK ||
            minterType == MinterType.FINANCIAL_INSTITUTION ||
            minterType == MinterType.OTHER,
            "Invalid minter type"
        );
        
        if (minterType == MinterType.BANK) {
            totalBankShare += sharePercentage;
        }
        
        require(
            totalBankShare >= MIN_BANK_SHARE,
            "Bank share must be at least 51%"
        );
        
        minters[minter] = MinterInfo({
            minter: minter,
            minterType: minterType,
            sharePercentage: sharePercentage,
            authorizedAmount: maxAmount,
            mintedAmount: 0,
            isAuthorized: true
        });
        
        minterList.push(minter);
        
        emit MinterRegistered(minter, minterType, sharePercentage);
    }
    
    function verifyBankMajority() 
        external 
        view 
        returns (bool) 
    {
        return totalBankShare >= MIN_BANK_SHARE;
    }
}
```

**구현 요건**:

contract BankOfKoreaMonitoring {
    address public bankOfKorea; // 한국은행 주소
    uint256 public issuanceLimit; // 발행 한도
    uint256 public currentIssuance; // 현재 발행량
    
    struct MonitoringData {
        uint256 timestamp;
        uint256 totalSupply;
        uint256 totalReserve;
        uint256 issuanceRate; // 발행 속도
        bool isWithinLimit;
    }
    
    MonitoringData[] public monitoringHistory;
    
    modifier onlyBankOfKorea() {
        require(msg.sender == bankOfKorea, "Only Bank of Korea");
        _;
    }
    
    function setIssuanceLimit(uint256 limit) external onlyBankOfKorea {
        issuanceLimit = limit;
        emit IssuanceLimitSet(limit);
    }
    
    function updateIssuance(uint256 newIssuance) external onlyAuthorized {
        currentIssuance = newIssuance;
        
        require(
            currentIssuance <= issuanceLimit,
            "Issuance exceeds limit"
        );
        
        monitoringHistory.push(MonitoringData({
            timestamp: block.timestamp,
            totalSupply: newIssuance,
            totalReserve: getTotalReserve(),
            issuanceRate: calculateIssuanceRate(),
            isWithinLimit: currentIssuance <= issuanceLimit
        }));
        
        emit IssuanceUpdated(newIssuance, issuanceLimit);
    }
    
    function getTotalReserve() internal view returns (uint256) {
        // 준비금 조회 로직
        return IReserveManager(reserveManagerAddress).getTotalReserve();
    }
    
    function calculateIssuanceRate() internal view returns (uint256) {
        // 발행 속도 계산 (시간당)
        if (monitoringHistory.length < 2) return 0;
        
        MonitoringData memory latest = monitoringHistory[monitoringHistory.length - 1];
        MonitoringData memory previous = monitoringHistory[monitoringHistory.length - 2];
        
        uint256 timeDiff = latest.timestamp - previous.timestamp;
        if (timeDiff == 0) return 0;
        
        uint256 supplyDiff = latest.totalSupply - previous.totalSupply;
        return (supplyDiff * 1 hours) / timeDiff;
    }
    
    function requestIssuanceIncrease(uint256 requestedAmount) 
        external 
        onlyAuthorized 
    {
        require(
            currentIssuance + requestedAmount <= issuanceLimit,
            "Would exceed limit"
        );
        
        // 한국은행 승인 요청
        emit IssuanceIncreaseRequested(requestedAmount);
    }
}
```

        emit ReserveUpdated(totalReserve, totalSupply);
    }
    
    function getReserveRatio() external view returns (uint256) {
        if (totalSupply == 0) return 0;
        return (totalReserve * 100) / totalSupply;
    }
    
    function verifyReserve() external view returns (bool) {
        return totalReserve >= totalSupply;
    }
}
```

#### A.3 Redemption Manager

**목적**: 액면가 1:1 상환

**기술 구현**:

```solidity
// RedemptionManager.sol
contract RedemptionManager {
    struct RedemptionRequest {
        address requester;
        uint256 amount;
        uint256 requestDate;
        uint256 deadline;
        bool isProcessed;
    }
    
    RedemptionRequest[] public redemptionQueue;
    
    function requestRedemption(uint256 amount) external {
        require(amount > 0, "Amount must be positive");
        
        IStablecoin(stablecoinAddress).burnFrom(msg.sender, amount);
        
        uint256 deadline = calculateDeadline();
        
        redemptionQueue.push(RedemptionRequest({
            requester: msg.sender,
            amount: amount,
            requestDate: block.timestamp,
            deadline: deadline,
            isProcessed: false
        }));
        
        emit RedemptionRequested(msg.sender, amount);
    }
    
    function processRedemption(uint256 requestId) external onlyAuthorized {
        RedemptionRequest storage request = redemptionQueue[requestId];
        require(!request.isProcessed, "Already processed");
        require(block.timestamp <= request.deadline, "Deadline passed");
        
        require(
            address(this).balance >= request.amount,
            "Insufficient reserve"
        );
        
        payable(request.requester).transfer(request.amount);
        request.isProcessed = true;
        
        emit RedemptionProcessed(requestId, request.requester, request.amount);
    }
    
    function calculateDeadline() internal view returns (uint256) {
        // 국가별 상환 기한 계산
        // 일본: 명시 없음
        // 싱가포르: 5영업일
        // 홍콩: 1영업일
        // 영국: 익영업일 말
        // 미국: 명시 없음
        return block.timestamp + 1 days; // 기본값
    }
}
```

        blacklisted[account] = true;
        kycVerified[account] = false;
        emit AccountBlacklisted(account);
    }

        external 
        view

               kycVerified[to] && 
               !blacklisted[from] && 
               !blacklisted[to];
    }
